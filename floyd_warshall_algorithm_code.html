Floyd-Warshall Algorithm in C++

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

void FloydWarshall(vector<vector<int>>& W, int n) {
    vector<vector<int>> D = W; // Initialize D as the weight matrix W

    for (int k = 0; k < n; ++k) {
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < n; ++j) {
                D[i][j] = min(D[i][j], D[i][k] + D[k][j]);
            }
        }
    }

    // Print the distance matrix D
    cout << "Distance matrix of the shortest paths:\n";
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n; ++j) {
            if (D[i][j] == 1e9) {
                cout << "INF ";
            } else {
                cout << D[i][j] << " ";
            }
        }
        cout << endl;
    }
}

int main() {
    int n = 4; // Number of vertices in the graph
    vector<vector<int>> W = {
        {0, 3, 1, 5},
        {2, 0, 4, 2},
        {3, 1, 0, 6},
        {4, 2, 3, 0}
    };

    FloydWarshall(W, n);

    return 0;
}
