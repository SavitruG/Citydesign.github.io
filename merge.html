MergeSort Algorithm in C++

#include <iostream>
#include <vector>

using namespace std;

// Merge function to merge two sorted arrays into one
void Merge(vector<int>& B, vector<int>& C, vector<int>& A) {
    int i = 0, j = 0, k = 0;
    int p = B.size(), q = C.size();

    // Merge the two arrays while both arrays have elements
    while (i < p && j < q) {
        if (B[i] <= C[j]) {
            A[k++] = B[i++];
        } else {
            A[k++] = C[j++];
        }
    }

    // If elements remain in B, copy them to A
    while (i < p) {
        A[k++] = B[i++];
    }

    // If elements remain in C, copy them to A
    while (j < q) {
        A[k++] = C[j++];
    }
}

// MergeSort function to recursively sort the array
void MergeSort(vector<int>& A) {
    int n = A.size();
    if (n > 1) {
        // Split the array into two halves
        int mid = n / 2;
        vector<int> B(A.begin(), A.begin() + mid); // First half
        vector<int> C(A.begin() + mid, A.end());   // Second half

        // Recursively sort both halves
        MergeSort(B);
        MergeSort(C);

        // Merge the two sorted halves into the original array A
        Merge(B, C, A);
    }
}

// Function to print the array
void PrintArray(const vector<int>& A) {
    for (int i = 0; i < A.size(); ++i) {
        cout << A[i] << " ";
    }
    cout << endl;
}

int main() {
    vector<int> A = {12, 11, 13, 5, 6, 7};

    cout << "Original Array: ";
    PrintArray(A);

    // Sorting the array using MergeSort
    MergeSort(A);

    cout << "Sorted Array: ";
    PrintArray(A);

    return 0;
}
