Priority Queue in C++

#include <iostream>
#include <queue>

using namespace std;

int main() {
    // Declaring a priority queue
    priority_queue<int> pq;
    
    // Inserting elements into the priority queue
    pq.push(10);
    pq.push(30);
    pq.push(20);
    pq.push(5);
    
    // Printing the top element (max element in default priority queue)
    cout << "Top element (Max element): " << pq.top() << endl;
    
    // Removing the top element
    pq.pop();
    
    cout << "After popping, new top element: " << pq.top() << endl;
    
    // Removing all elements
    cout << "Priority Queue elements: " << endl;
    while (!pq.empty()) {
        cout << pq.top() << " ";  // Displaying the top element
        pq.pop();  // Popping the top element
    }
    
    return 0;
}
