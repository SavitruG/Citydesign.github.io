Hashing

#include <iostream>
#include <unordered_map>

using namespace std;

int main() {
    unordered_map<string, int> hashMap;

    // Inserting elements into the hash map
    hashMap["Alice"] = 25;
    hashMap["Bob"] = 30;
    hashMap["Charlie"] = 35;

    // Accessing values by keys
    cout << "Alice's age: " << hashMap["Alice"] << endl;
    cout << "Bob's age: " << hashMap["Bob"] << endl;
    cout << "Charlie's age: " << hashMap["Charlie"] << endl;

    // Searching for a key
    if (hashMap.find("David") != hashMap.end()) {
        cout << "David's age: " << hashMap["David"] << endl;
    } else {
        cout << "David not found in the hash map." << endl;
    }

    return 0;
}
