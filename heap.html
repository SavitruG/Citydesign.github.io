Heap in C++

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

void printHeap(vector<int>& heap) {
    for (int val : heap)
        cout << val << " ";
    cout << endl;
}

int main() {
    vector<int> heap = {10, 20, 15, 30, 40};
    make_heap(heap.begin(), heap.end());
    
    cout << "Max Heap: ";
    printHeap(heap);
    
    heap.push_back(50);
    push_heap(heap.begin(), heap.end());
    
    cout << "Heap after push: ";
    printHeap(heap);
    
    pop_heap(heap.begin(), heap.end());
    heap.pop_back();
    
    cout << "Heap after pop: ";
    printHeap(heap);
    
    return 0;
}
